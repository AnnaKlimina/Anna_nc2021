let i = 0;

setTimeout(() => alert(i), 100); 

// предположим, что время выполнения этой функции >100 мс
for(let j = 0; j < 100000000; j++) {
  i++;
}

вызов выполнится после цикла, так как любой вызов setTimeout 
выполнется после завершени текущего кода.
после цикла функция возьмет значение i из внешнего лексического окружения,
т.е. i = 100000000
